class Persona {

    /**
     * codigo único para la persona, autogenerado por el gestor de la base de datos Mysql.
     * @type {int}
     */
    codigo;
    /**
     * nombres de la persona
     * @type {string}
     */
    nombres;
    /**
     * apellidos de la persona
     * @type {string}
     */
    apellidos;
     /**
     * identificación de la persona
     * @type {string}
     */
    identificacion;
    /**
     * teléfono de la persona
     * @type {string}
     */
    telefono;
    /**
     * email de la persona
     * @type {string}
     */
    email;
    /**
     * código de usuario, el cual se relaciona la persona con un usuario para iniciar sesión
     * @type {int}
     */
    codigo_usuario;
    /**
     * tipo de persona, para representar si la persona es un admin o un cliente
     * @type {string}
     */
    tipo;

    /**
     * 
     * @param {string} nombres - Nombres de la persona
     * @param {string} apellidos - Apellidos de la persona
     * @param {string} identificacion - Identificación de la persona
     * @param {string} telefono - Teléfono de la persona
     * @param {string} email - Email de la persona
     * @param {int} codigo_usuario - Código de usuario con el cual se relaciona con un usuario
     * @param {string} tipo - Tipo de persona (Administrador o cliente)
     */
    constructor (nombres, apellidos, identificacion, telefono, email, codigo_usuario, tipo) {
        this.nombres = nombres;
        this.apellidos = apellidos;
        this.identificacion = identificacion;
        this.telefono = telefono;
        this.email = email;
        this.codigo_usuario = codigo_usuario;
        this.tipo = tipo;
    }


}

module.exports= Persona;
